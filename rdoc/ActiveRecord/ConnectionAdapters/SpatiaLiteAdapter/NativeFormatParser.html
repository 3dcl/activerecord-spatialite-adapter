def parse(data_)
  if data_[0,1] =~ %r[0-9a-fA-F]/
    data_ = [data_].pack('H*')
  end
  @little_endian = data_[1,1] == "\x01"
  srid_ = data_[2,4].unpack(@little_endian ? 'V' : 'N').first
  begin
    _start_scanner(data_)
    obj_ = _parse_object(false)
    _get_byte(0xfe)
  ensure
    _clean_scanner
  end
  obj_
end
